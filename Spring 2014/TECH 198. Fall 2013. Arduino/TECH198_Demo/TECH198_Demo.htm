Sample code for Tech 198 Arduino Library
1. Declare Library
2. Select four pins (must be pins with capable of PWM)
3. Syntax for motor() function:
    motor( <motorSelect> , <motorDirection> , <motorSpeed> );
    motor( <0 or 1>, <0 or 1>, <0 to 255> );
    
Examples:
motor(0,0,0); //First motor, forward, speed = 0 out of 255;
motor(0,1,128); //First motor, reverse, speed = 128 out of 255;
motor(1,0,255); //Second motor, forward, speed - 255 out of 255;
*/

//Declaring Library
#include <Tech198.h>

//Motor Pins
//Motor 1 signals connected to pins 3 and 5
//Motor 2 signals connected to pins 6 and 9
Tech198 robot(3,5,6,9);

void setup()
{
}

void loop()
{
  //Run motor 1
  robot.motor(0,1,128); //Forwards
  delay(2000);
  robot.motor(0,0,128); //Backwards
  delay(2000);
  robot.motor(0,0,0); //Stop
  delay(1000);
  
  //Run motor 2
  robot.motor(1,1,128); //Forwards
  delay(2000);
  robot.motor(1,0,128); //Backwards
  delay(2000);
  robot.motor(1,0,0); //Stop
  delay(1000);
}
